	desc = TTD_Indian_war_news.1.d
	picture = GFX_news_event_india_wars

	is_triggered_only = yes

	option = {
		name = TTD_Indian_war_news.1.a
		if = {
			limit = { is_ai = no }

				every_country = {
					limit = { is_ai = no }
					set_temp_variable = { TNO_temp_super_event = token:SE_INDIAN_WARS }
					TNO_fire_super_event = yes
				}
			}
		}
	}
}
